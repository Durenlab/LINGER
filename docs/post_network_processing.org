* Post networks processing

- reg_file is your raw TF-TG networks file (either population network or cell type specific network)
- tf_tg_df is normalized networks matrix
- tf_tg_melt is your processed long table of TF-TG pairs

#+BEGIN_SRC python
# read and normalize LINGER result

reg_file = "/path/to/your/LINGER/output/networks/" # something like "output/cell_population_TF_RE_binding.txt" or "/output/cell_type_specific_trans_regulatory_XXX.txt"
reg = pd.read_csv(reg_file, sep='\t', index_col=0)
trans_reg = reg.copy()
TFset=trans_reg.columns
TGset=trans_reg.index
R1 = stats.zscore(trans_reg,1);R1[np.isnan(R1)] = 0.0
R2 = stats.zscore(trans_reg,0);R2[np.isnan(R2)] = 0.0
Z=(R1+R2)/2
tf_tg_df= pd.DataFrame(Z,index=TGset,columns=TFset)
#+END_SRC
