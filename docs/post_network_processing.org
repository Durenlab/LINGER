# normalize LINGER result

reg_file = "/path/to/your/LINGER/output/networks/" # something like "outputcell_population_TF_RE_binding.txt" or "/output/cell_type_specific_trans_regulatory_XXX.txt"
reg = pd.read_csv(reg_file, sep='\t', index_col=0)
trans_reg = reg.copy()
TFset=trans_reg.columns
TGset=trans_reg.index
R1 = stats.zscore(trans_reg,1);R1[np.isnan(R1)] = 0.0
R2 = stats.zscore(trans_reg,0);R2[np.isnan(R2)] = 0.0
Z=R1+R2

# you network threshold ()
threshold = 0
Z[Z<threshold]=0
tf_tg_df= pd.DataFrame(Z,index=TGset,columns=TFset)


# tf_tg_df: Index=TF, Columns=Gene (For generaging long table)
tf_tg_melt = tf_tg_df.reset_index().melt(id_vars='index', var_name='TargetGene', value_name='Score')
tf_tg_melt.rename(columns={'index': 'TF'}, inplace=True)
tf_tg_melt = tf_tg_melt[tf_tg_melt['Score'] != 0] # Filter out zero scores
